    // ── WRITE TO FIRESTORE ────────────────

    // Owner dashboard gets this in real time via onSnapshot listener

    const docRef = await addDoc(collection(db, 'orders'), order);

    currentOrderToken = token;


    // Show success modal

    document.getElementById('m-tok').textContent = token;

    document.getElementById('m-sub').textContent =

      `${items.map(i=>i.name+' ×'+i.qty).join(', ')} · ₹${grand} · ${payment.toUpperCase()}`;

    document.getElementById('modal').classList.add('show');


    // Live-track this order's status from Firestore

    startLiveTracking(docRef.id);


    // Reset cart

    cart = {};

    renderMenu();

    updateSummary();

    document.getElementById('ci-name').value = '';

    document.getElementById('ci-phone').value = '';

    document.getElementById('ci-notes').value = '';

  } catch(err) {

    alert('Could not place order — please check your connection and try again.\n\n' + err.message);

    console.error(err);

  }


  btn.disabled = false;

  btn.textContent = '🔥 Place Order';

}


// ── LIVE ORDER TRACKING ─────────────────

// Listens to Firestore doc in real time —

// when owner marks "Preparing/Ready/Done",

// the customer's tracking dots update instantly

function startLiveTracking(docId){

  if(unsubscribeTracker) unsubscribeTracker();

  // Reset dots

  for(let i=0;i<4;i++){

    const d = document.getElementById('ts'+i);

    if(d) d.className = 'ts-dot' + (i===0?' done':'');

  }

  const orderRef = doc(db, 'orders', docId);

  unsubscribeTracker = onSnapshot(orderRef, snap => {

    if(!snap.exists()) return;

    const s = snap.data().status;

    const stepMap = {new:0, preparing:1, ready:2, done:3};

    const cur = stepMap[s] ?? 0;

    for(let i=0;i<4;i++){

      const d = document.getElementById('ts'+i);

      if(d) d.className = 'ts-dot' + (i<cur?' done':i===cur?' act':'');

    }

  });

}


function closeModal(){

  document.getElementById('modal').classList.remove('show');

  // Keep tracking running so customer can re-open and still see status

}
